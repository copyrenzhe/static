默认

    
$("#paging").pagination({
  currentPage: 1,
  total:470,
});
    
  
    
$("#paging11").pagination();

设置边缘

    
$("#paging2").pagination({
  currentPage: 1,
  total:470,
  edges:0,
  displayEdges:0
});
    
  
  
$("#paging3").pagination({
  currentPage: 1,
  total:470,
  edges:2,
  displayEdges:0
});

设置显示个数

    
$("#paging4").pagination({
  currentPage: 1,
  total:470,
  edges:0,
  displayEdges:2
});
    
  
    
$("#paging5").pagination({

$("#paging6").pagination({
  currentPage: 1,
  total:470,

});

$("#paging7").pagination({
  currentPage: 1,
  total:470,

});

允许选择页大小

    
$("#paging8").pagination({
  currentPage: 1,
  total:470,
  pageSizeSelect:true,
  showTotal:true
});

pageChange事件

    
$("#paging9").pagination({
  currentPage: 1,
  total:470
}).on('pageChange',function(e){
  $("#console").append('pageChange:page='+e.page);
});
